-- IoTBay.Courier definition

CREATE TABLE `Courier` (
  `CourierID` int NOT NULL,
  `CourierName` varchar(255) NOT NULL,
  `CourierFees` float NOT NULL,
  PRIMARY KEY (`CourierID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;


-- IoTBay.Customer definition

CREATE TABLE `Customer` (
  `Customer_ID` int NOT NULL AUTO_INCREMENT COMMENT 'Customer''s unique ID, beginning at "CU-1000". Increase by one for each cutomer',
  `Given_Name` varchar(15) NOT NULL COMMENT 'Customer''s  given name',
  `Family_Name` varchar(15) NOT NULL COMMENT 'Customer''s family name',
  `Email` varchar(30) NOT NULL COMMENT 'Customer''s email in the format, max 15 characters, @, max 15 characters (to prevent false emails.',
  `Password` varchar(20) NOT NULL COMMENT 'Customer''s secure password',
  `DOB` date NOT NULL COMMENT 'Customer''s date of birth',
  `Phone_Number` varchar(10) NOT NULL COMMENT 'Customer''s phone number',
  PRIMARY KEY (`Customer_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;


-- IoTBay.Staff definition

CREATE TABLE `Staff` (
  `Staff_ID` int NOT NULL AUTO_INCREMENT,
  `Given_Name` varchar(15) NOT NULL,
  `Family_Name` varchar(15) NOT NULL,
  `Email` varchar(30) NOT NULL,
  `Password` varchar(20) NOT NULL,
  `DOB` date NOT NULL,
  `Admin_Role` int NOT NULL,
  PRIMARY KEY (`Staff_ID`)
) ENGINE=InnoDB AUTO_INCREMENT=2 DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;


-- IoTBay.Address definition

CREATE TABLE `Address` (
  `Address_ID` int NOT NULL AUTO_INCREMENT,
  `Customer_ID` int NOT NULL,
  `Unit_Number` varchar(10) CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci DEFAULT NULL,
  `Street_Number` varchar(10) CHARACTER SET utf8mb4 COLLATE utf8mb4_0900_ai_ci NOT NULL,
  `Street_Name` varchar(60) NOT NULL,
  `City_Name` varchar(40) NOT NULL,
  `State_Name` varchar(20) NOT NULL,
  `Post_Code` int NOT NULL,
  `Country_Name` varchar(20) NOT NULL,
  PRIMARY KEY (`Address_ID`),
  KEY `Address_Customer_FK` (`Customer_ID`),
  CONSTRAINT `Address_Customer_FK` FOREIGN KEY (`Customer_ID`) REFERENCES `Customer` (`Customer_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;


-- IoTBay.`Order` definition

CREATE TABLE `Order` (
  `Order_ID` int NOT NULL AUTO_INCREMENT,
  `Customer_ID` int NOT NULL,
  `Order_Date` date NOT NULL,
  `Status` varchar(256) DEFAULT NULL,
  PRIMARY KEY (`Order_ID`),
  KEY `Order_Customer_FK` (`Customer_ID`),
  CONSTRAINT `Order_Customer_FK` FOREIGN KEY (`Customer_ID`) REFERENCES `Customer` (`Customer_ID`)
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COLLATE=utf8mb4_0900_ai_ci;
